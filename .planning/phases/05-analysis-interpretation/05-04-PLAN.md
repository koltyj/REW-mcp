/**
 * Tool: rew.analyze_room
 *
 * Unified room analysis with plain language interpretation and prioritized recommendations.
 * Combines room modes, peaks/nulls, sub integration, and L/R symmetry analysis.
 */

import { z } from 'zod';
import { measurementStore } from '../store/measurement.js';
import type { ToolResponse, RoomDimensions, ConfidenceLevel, Severity } from '../types/index.js';
import { detectPeaks, detectNulls } from '../analysis/peaks-nulls.js';
import { calculateTheoreticalModes } from '../analysis/room-modes.js';
import { analyzeSubIntegration } from '../analysis/sub-integration.js';
import {
  interpretRoomModes,
  interpretPeaksNulls,
  interpretSubIntegration,
  interpretLRSymmetry,
  prioritizeIssues,
  type Recommendation,
  type IssueInput
} from '../interpretation/index.js';

// Input schema
export const AnalyzeRoomInputSchema = z.object({
  measurement_id: z.string().describe('Primary measurement to analyze'),
  left_measurement_id: z.string().optional().describe('Left speaker measurement for L/R symmetry'),
  right_measurement_id: z.string().optional().describe('Right speaker measurement for L/R symmetry'),
  sub_measurement_id: z.string().optional().describe('Subwoofer measurement for integration analysis'),
  mains_measurement_id: z.string().optional().describe('Mains measurement for sub integration (uses primary if not specified)'),
  room_dimensions: z.object({
    length: z.number().positive(),
    width: z.number().positive(),
    height: z.number().positive()
  }).optional().describe('Room dimensions in feet for mode correlation'),
  crossover_hz: z.number().min(40).max(200).optional().describe('Crossover frequency for sub analysis'),
  sub_delay_ms: z.number().optional().describe('Current subwoofer delay in ms'),
  sub_polarity: z.enum(['normal', 'inverted']).optional().describe('Current subwoofer polarity')
});

export type AnalyzeRoomInput = z.infer<typeof AnalyzeRoomInputSchema>;

export interface RoomAnalysisResult {
  // Overall summary
  overall_summary: string;
  overall_severity: Severity;
  overall_confidence: ConfidenceLevel;

  // Prioritized recommendations (ANLZ-07)
  prioritized_recommendations: Recommendation[];

  // Individual analyses
  peaks_nulls_analysis?: {
    summary: string;
    peak_count: number;
    null_count: number;
    sbir_null_count: number;
    recommendations: Recommendation[];
  };

  room_modes_analysis?: {
    summary: string;
    mode_count: number;
    correlated_peaks: number;
    recommendations: Recommendation[];
  };

  sub_integration_analysis?: {
    summary: string;
    phase_alignment_quality: string;
    phase_inversion_detected: boolean;
    inversion_recommended: boolean;
    timing_adjustment_ms: number;
    recommendations: Recommendation[];
  };

  lr_symmetry_analysis?: {
    summary: string;
    overall_rating: string;
    overall_deviation_db: number;
    asymmetric_bands: string[];
    recommendations: Recommendation[];
  };

  // Raw data for programmatic use
  analysis_confidence: ConfidenceLevel;
  analysis_limitations: string[];
}

/**
 * Execute unified room analysis
 */
export async function executeAnalyzeRoom(input: AnalyzeRoomInput): Promise<ToolResponse<RoomAnalysisResult>> {
  try {
    const validated = AnalyzeRoomInputSchema.parse(input);

    // Get primary measurement
    const primaryMeasurement = measurementStore.get(validated.measurement_id);
    if (!primaryMeasurement) {
      return {
        status: 'error',
        error_type: 'measurement_not_found',
        message: `Measurement '${validated.measurement_id}' not found`,
        suggestion: 'Ingest measurement data first using rew.ingest_measurement'
      };
    }

    const allRecommendations: Recommendation[] = [];
    const allIssues: IssueInput[] = [];
    const limitations: string[] = [];
    let overallSeverity: Severity = 'negligible';

    // ===== 1. Peaks and Nulls Analysis =====
    const peaks = detectPeaks(primaryMeasurement.frequency_response, {
      threshold_db: 5,
      min_frequency_hz: 20,
      max_frequency_hz: 500 // Focus on bass/low-mid for room analysis
    });
    const nulls = detectNulls(primaryMeasurement.frequency_response, {
      threshold_db: -6,
      min_frequency_hz: 20,
      max_frequency_hz: 500
    });

    const peaksNullsInterp = interpretPeaksNulls(peaks, nulls);
    allRecommendations.push(...peaksNullsInterp.recommendations);

    // Collect issues for prioritization
    for (const rec of peaksNullsInterp.recommendations) {
      allIssues.push({
        issue: rec.action,
        severity: peaksNullsInterp.severity,
        fixability: rec.fixability,
        category: rec.category
      });
    }

    if (severityRank(peaksNullsInterp.severity) > severityRank(overallSeverity)) {
      overallSeverity = peaksNullsInterp.severity;
    }

    // ===== 2. Room Modes Analysis =====
    let roomModesResult: RoomAnalysisResult['room_modes_analysis'] | undefined;

    if (validated.room_dimensions) {
      const modes = calculateTheoreticalModes(validated.room_dimensions, { max_frequency_hz: 200 });
      const roomModesInterp = interpretRoomModes(modes, peaks, validated.room_dimensions);

      roomModesResult = {
        summary: roomModesInterp.summary,
        mode_count: roomModesInterp.data.modes.length,
        correlated_peaks: roomModesInterp.data.correlations.length,
        recommendations: roomModesInterp.recommendations
      };

      allRecommendations.push(...roomModesInterp.recommendations);

      for (const rec of roomModesInterp.recommendations) {
        allIssues.push({
          issue: rec.action,
          severity: roomModesInterp.severity,
          fixability: rec.fixability,
          category: rec.category
        });
      }

      if (severityRank(roomModesInterp.severity) > severityRank(overallSeverity)) {
        overallSeverity = roomModesInterp.severity;
      }
    } else {
      limitations.push('Room dimensions not provided - mode correlation not available');
    }

    // ===== 3. Sub Integration Analysis =====
    let subIntegrationResult: RoomAnalysisResult['sub_integration_analysis'] | undefined;

    if (validated.sub_measurement_id) {
      const subMeasurement = measurementStore.get(validated.sub_measurement_id);
      if (!subMeasurement) {
        return {
          status: 'error',
          error_type: 'measurement_not_found',
          message: `Sub measurement '${validated.sub_measurement_id}' not found`
        };
      }

      const mainsMeasurement = validated.mains_measurement_id
        ? measurementStore.get(validated.mains_measurement_id)
        : primaryMeasurement;

      if (!mainsMeasurement) {
        return {
          status: 'error',
          error_type: 'measurement_not_found',
          message: `Mains measurement '${validated.mains_measurement_id}' not found`
        };
      }

      const subAnalysis = analyzeSubIntegration(
        mainsMeasurement.frequency_response,
        subMeasurement.frequency_response,
        {
          crossover_hz: validated.crossover_hz,
          current_sub_delay_ms: validated.sub_delay_ms,
          current_sub_polarity: validated.sub_polarity
        }
      );

      const subInterp = interpretSubIntegration(subAnalysis);

      subIntegrationResult = {
        summary: subInterp.summary,
        phase_alignment_quality: subInterp.data.crossover.phase_alignment_quality,
        phase_inversion_detected: subInterp.data.phase_inversion.is_phase_inverted,
        inversion_recommended: subInterp.data.phase_inversion.inversion_recommended,
        timing_adjustment_ms: subInterp.data.timing.adjustment_needed_ms,
        recommendations: subInterp.recommendations
      };

      allRecommendations.push(...subInterp.recommendations);

      for (const rec of subInterp.recommendations) {
        allIssues.push({
          issue: rec.action,
          severity: subInterp.severity,
          fixability: rec.fixability,
          category: rec.category
        });
      }

      if (severityRank(subInterp.severity) > severityRank(overallSeverity)) {
        overallSeverity = subInterp.severity;
      }
    }

    // ===== 4. L/R Symmetry Analysis =====
    let lrSymmetryResult: RoomAnalysisResult['lr_symmetry_analysis'] | undefined;

    if (validated.left_measurement_id && validated.right_measurement_id) {
      const leftMeasurement = measurementStore.get(validated.left_measurement_id);
      const rightMeasurement = measurementStore.get(validated.right_measurement_id);

      if (!leftMeasurement) {
        return {
          status: 'error',
          error_type: 'measurement_not_found',
          message: `Left measurement '${validated.left_measurement_id}' not found`
        };
      }
      if (!rightMeasurement) {
        return {
          status: 'error',
          error_type: 'measurement_not_found',
          message: `Right measurement '${validated.right_measurement_id}' not found`
        };
      }

      const lrInterp = interpretLRSymmetry(
        leftMeasurement.frequency_response,
        rightMeasurement.frequency_response
      );

      lrSymmetryResult = {
        summary: lrInterp.summary,
        overall_rating: lrInterp.data.overall_rating,
        overall_deviation_db: lrInterp.data.overall_deviation_db,
        asymmetric_bands: lrInterp.data.asymmetric_bands,
        recommendations: lrInterp.recommendations
      };

      allRecommendations.push(...lrInterp.recommendations);

      for (const rec of lrInterp.recommendations) {
        allIssues.push({
          issue: rec.action,
          severity: lrInterp.severity,
          fixability: rec.fixability,
          category: rec.category
        });
      }

      if (severityRank(lrInterp.severity) > severityRank(overallSeverity)) {
        overallSeverity = lrInterp.severity;
      }
    }

    // ===== 5. Prioritize All Recommendations =====
    const prioritizedIssues = prioritizeIssues(allIssues);

    // Assign priority numbers based on sorted order
    const prioritizedRecommendations = prioritizedIssues.map((issue, index) => ({
      ...issue.recommendation,
      priority: index + 1
    }));

    // Limit to top 5 recommendations
    const topRecommendations = prioritizedRecommendations.slice(0, 5);

    // ===== 6. Generate Overall Summary =====
    const summaryParts: string[] = [];

    if (overallSeverity === 'significant') {
      summaryParts.push('Room has significant acoustic issues requiring attention');
    } else if (overallSeverity === 'moderate') {
      summaryParts.push('Room has moderate acoustic issues that can be improved');
    } else if (overallSeverity === 'minor') {
      summaryParts.push('Room has minor acoustic issues');
    } else {
      summaryParts.push('Room acoustics are generally good');
    }

    if (topRecommendations.length > 0) {
      summaryParts.push(`Top priority: ${topRecommendations[0].action}`);
    }

    const overallSummary = summaryParts.join('. ') + '.';

    const result: RoomAnalysisResult = {
      overall_summary: overallSummary,
      overall_severity: overallSeverity,
      overall_confidence: primaryMeasurement.data_quality.confidence,
      prioritized_recommendations: topRecommendations,
      peaks_nulls_analysis: {
        summary: peaksNullsInterp.summary,
        peak_count: peaks.length,
        null_count: nulls.length,
        sbir_null_count: peaksNullsInterp.data.sbir_nulls.length,
        recommendations: peaksNullsInterp.recommendations
      },
      room_modes_analysis: roomModesResult,
      sub_integration_analysis: subIntegrationResult,
      lr_symmetry_analysis: lrSymmetryResult,
      analysis_confidence: primaryMeasurement.data_quality.confidence,
      analysis_limitations: limitations
    };

    return {
      status: 'success',
      data: result
    };

  } catch (error) {
    if (error instanceof z.ZodError) {
      return {
        status: 'error',
        error_type: 'validation_error',
        message: `Invalid input: ${error.errors.map(e => `${e.path.join('.')}: ${e.message}`).join(', ')}`
      };
    }

    return {
      status: 'error',
      error_type: 'internal_error',
      message: error instanceof Error ? error.message : 'Unknown error'
    };
  }
}

// Helper to rank severity for comparison
function severityRank(severity: Severity): number {
  const ranks: Record<Severity, number> = {
    negligible: 0,
    minor: 1,
    moderate: 2,
    significant: 3
  };
  return ranks[severity];
}
