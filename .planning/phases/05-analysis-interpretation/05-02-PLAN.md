```typescript
import type { TheoreticalMode, RoomDimensions, DetectedPeak, ConfidenceLevel, Severity } from '../types/index.js';
import type { InterpretedResult, Recommendation, IssueInput } from './types.js';

export interface RoomModesData {
  modes: TheoreticalMode[];
  correlations: Array<{
    mode: TheoreticalMode;
    peak: DetectedPeak;
    match_error_percent: number;
  }>;
  dimensions?: RoomDimensions;
}

export function interpretRoomModes(
  modes: TheoreticalMode[],
  peaks: DetectedPeak[],
  dimensions?: RoomDimensions
): InterpretedResult<RoomModesData> {
  // Find correlations between theoretical modes and detected peaks
  const correlations: RoomModesData['correlations'] = [];
  const correlatedPeaks = new Set<number>();

  for (const mode of modes) {
    // Find peaks within 5% of theoretical mode frequency
    for (const peak of peaks) {
      const errorPercent = Math.abs(peak.frequency_hz - mode.frequency_hz) / mode.frequency_hz * 100;
      if (errorPercent <= 5 && !correlatedPeaks.has(peak.frequency_hz)) {
        correlations.push({ mode, peak, match_error_percent: errorPercent });
        correlatedPeaks.add(peak.frequency_hz);
        break;
      }
    }
  }

  // Generate summary
  const axialCount = modes.filter(m => m.mode_type === 'axial').length;
  const tangentialCount = modes.filter(m => m.mode_type === 'tangential').length;
  const correlatedCount = correlations.length;

  let summary = `Found ${modes.length} theoretical room modes below 200 Hz`;
  if (modes.length > 0) {
    summary += ` (${axialCount} axial, ${tangentialCount} tangential).`;
  } else {
    summary += '.';
  }

  if (correlatedCount > 0) {
    summary += ` ${correlatedCount} detected peak${correlatedCount > 1 ? 's' : ''} correlate with room modes.`;
  }

  if (dimensions) {
    summary += ` Room dimensions: ${dimensions.length}' x ${dimensions.width}' x ${dimensions.height}'.`;
  } else {
    summary += ' Provide room dimensions for more accurate mode correlation.';
  }

  // Generate recommendations
  const recommendations: Recommendation[] = [];
  const issues: IssueInput[] = [];

  // Check for problematic mode clusters (modes within 5 Hz of each other)
  const sortedModes = [...modes].sort((a, b) => a.frequency_hz - b.frequency_hz);
  for (let i = 0; i < sortedModes.length - 1; i++) {
    const gap = sortedModes[i + 1].frequency_hz - sortedModes[i].frequency_hz;
    if (gap < 5) {
      issues.push({
        issue: `Mode cluster at ${sortedModes[i].frequency_hz.toFixed(0)}-${sortedModes[i + 1].frequency_hz.toFixed(0)} Hz`,
        severity: 'moderate',
        fixability: 'placement',
        category: 'room_modes'
      });
    }
  }

  // Severe correlations get recommendations
  for (const corr of correlations) {
    if (corr.peak.severity === 'significant' || corr.peak.severity === 'moderate') {
      recommendations.push({
        action: `Address ${corr.mode.mode_type} mode at ${corr.mode.frequency_hz.toFixed(0)} Hz (${corr.mode.dimension} dimension)`,
        expected_impact: 'Repositioning may reduce modal impact by 3-6 dB',
        priority: corr.peak.severity === 'significant' ? 1 : 2,
        fixability: 'placement',
        category: 'room_modes'
      });
    }
  }

  // Determine overall severity
  const significantCorrelations = correlations.filter(c => c.peak.severity === 'significant').length;
  const moderateCorrelations = correlations.filter(c => c.peak.severity === 'moderate').length;

  let severity: Severity = 'negligible';
  if (significantCorrelations >= 2) severity = 'significant';
  else if (significantCorrelations >= 1 || moderateCorrelations >= 3) severity = 'moderate';
  else if (moderateCorrelations >= 1) severity = 'minor';

  // Confidence based on dimension availability
  const confidence: ConfidenceLevel = dimensions ? 'high' : 'medium';

  return {
    data: { modes, correlations, dimensions },
    summary,
    recommendations,
    severity,
    confidence
  };
}
```

```typescript
import type { DetectedPeak, DetectedNull, ConfidenceLevel, Severity } from '../types/index.js';
import type { InterpretedResult, Recommendation, IssueInput } from './types.js';

const SPEED_OF_SOUND_FT_S = 1125;

export interface SBIRClassification {
  is_sbir: boolean;
  confidence: ConfidenceLevel;
  estimated_boundary_distance_ft?: number;
  boundary_type?: 'rear_wall' | 'front_wall' | 'side_wall' | 'floor' | 'ceiling' | 'unknown';
  explanation: string;
}

export interface PeaksNullsData {
  peaks: DetectedPeak[];
  nulls: DetectedNull[];
  sbir_nulls: Array<DetectedNull & { sbir: SBIRClassification }>;
  glm_addressable_peaks: number;
  unaddressable_nulls: number;
}

/**
 * Classify whether a null is likely SBIR based on frequency range, Q factor, and calculated distance
 * SBIR (Speaker Boundary Interference Response) occurs when direct and reflected sound cancel
 * Formula: distance_ft = speed_of_sound / (4 * frequency_hz)
 */
export function classifySBIR(null_: DetectedNull): SBIRClassification {
  // SBIR nulls are typically:
  // - Above 60 Hz (below that is room modes)
  // - Below 300 Hz (above that is less likely boundary interference)
  // - High Q (narrow nulls from single reflection)

  if (null_.frequency_hz < 60) {
    return {
      is_sbir: false,
      confidence: 'high',
      explanation: `Frequency ${null_.frequency_hz.toFixed(0)} Hz is below 60 Hz - likely room mode rather than SBIR`
    };
  }

  if (null_.frequency_hz > 300) {
    return {
      is_sbir: false,
      confidence: 'high',
      explanation: `Frequency ${null_.frequency_hz.toFixed(0)} Hz is above 300 Hz - outside typical SBIR range`
    };
  }

  if (null_.q_factor < 5) {
    return {
      is_sbir: false,
      confidence: 'medium',
      explanation: `Wide null (Q=${null_.q_factor.toFixed(1)}) suggests room mode rather than boundary interference`
    };
  }

  // Calculate boundary distance from null frequency
  // Formula: cancellation_frequency = speed_of_sound / (2 * path_length_difference)
  // Path length difference = 2 * distance_to_boundary
  // Therefore: distance = speed_of_sound / (4 * cancellation_frequency)
  const quarterWavelength = SPEED_OF_SOUND_FT_S / (4 * null_.frequency_hz);

  // SBIR typically occurs when speaker is 1-4 ft from boundary
  const is_sbir = quarterWavelength >= 1 && quarterWavelength <= 4;

  if (!is_sbir) {
    return {
      is_sbir: false,
      confidence: 'medium',
      estimated_boundary_distance_ft: quarterWavelength,
      explanation: `Calculated distance (${quarterWavelength.toFixed(1)} ft) outside typical SBIR range (1-4 ft)`
    };
  }

  return {
    is_sbir: true,
    confidence: 'high',
    estimated_boundary_distance_ft: quarterWavelength,
    boundary_type: 'unknown', // Would need position data to determine which boundary
    explanation: `Narrow null at ${null_.frequency_hz.toFixed(0)} Hz suggests speaker is ~${quarterWavelength.toFixed(1)} ft from a boundary`
  };
}

export function interpretPeaksNulls(
  peaks: DetectedPeak[],
  nulls: DetectedNull[]
): InterpretedResult<PeaksNullsData> {
  // Classify SBIR for all nulls
  const sbirNulls: PeaksNullsData['sbir_nulls'] = [];
  for (const null_ of nulls) {
    const sbir = classifySBIR(null_);
    if (sbir.is_sbir) {
      sbirNulls.push({ ...null_, sbir });
    }
  }

  // Count GLM-addressable peaks
  const glmAddressablePeaks = peaks.filter(p => p.glm_addressable).length;
  const unaddressableNulls = nulls.filter(n => !n.glm_addressable).length;

  // Generate summary
  let summary = '';

  if (peaks.length === 0 && nulls.length === 0) {
    summary = 'No significant peaks or nulls detected. Response is relatively smooth.';
  } else {
    const parts: string[] = [];

    if (peaks.length > 0) {
      const significantPeaks = peaks.filter(p => p.severity === 'significant').length;
      parts.push(`${peaks.length} peak${peaks.length > 1 ? 's' : ''} detected`);
      if (glmAddressablePeaks > 0) {
        parts.push(`${glmAddressablePeaks} addressable by GLM/EQ`);
      }
      if (significantPeaks > 0) {
        parts.push(`${significantPeaks} significant`);
      }
    }

    if (nulls.length > 0) {
      const significantNulls = nulls.filter(n => n.severity === 'significant').length;
      parts.push(`${nulls.length} null${nulls.length > 1 ? 's' : ''} detected`);
      if (significantNulls > 0) {
        parts.push(`${significantNulls} significant`);
      }
    }

    summary = parts.join('. ') + '.';

    if (sbirNulls.length > 0) {
      summary += ` ${sbirNulls.length} null${sbirNulls.length > 1 ? 's appear' : ' appears'} to be SBIR (boundary interference).`;
    }

    if (unaddressableNulls > 0) {
      summary += ` Note: ${unaddressableNulls} null${unaddressableNulls > 1 ? 's cannot' : ' cannot'} be fixed with EQ - repositioning required.`;
    }
  }

  // Generate recommendations
  const recommendations: Recommendation[] = [];

  // SBIR nulls get highest priority (placement fix)
  for (const sbirNull of sbirNulls) {
    if (sbirNull.severity === 'significant' || sbirNull.severity === 'moderate') {
      recommendations.push({
        action: `Move speaker ~${sbirNull.sbir.estimated_boundary_distance_ft?.toFixed(1)} ft from boundary to address ${sbirNull.frequency_hz.toFixed(0)} Hz null`,
        expected_impact: `Should reduce ${sbirNull.frequency_hz.toFixed(0)} Hz null by 3-10 dB`,
        priority: sbirNull.severity === 'significant' ? 1 : 2,
        fixability: 'placement',
        category: 'sbir'
      });
    }
  }

  // Significant peaks
  const significantPeaks = peaks.filter(p => p.severity === 'significant');
  for (const peak of significantPeaks) {
    if (peak.glm_addressable) {
      recommendations.push({
        action: `Apply EQ cut at ${peak.frequency_hz.toFixed(0)} Hz (${peak.deviation_db.toFixed(1)} dB peak)`,
        expected_impact: `GLM/EQ can reduce this peak`,
        priority: 2,
        fixability: 'settings',
        category: 'peak'
      });
    } else {
      recommendations.push({
        action: `Consider acoustic treatment or repositioning to address ${peak.frequency_hz.toFixed(0)} Hz peak`,
        expected_impact: 'May reduce peak impact',
        priority: 3,
        fixability: 'treatment',
        category: 'peak'
      });
    }
  }

  // Determine overall severity
  const significantIssues =
    peaks.filter(p => p.severity === 'significant').length +
    nulls.filter(n => n.severity === 'significant').length;
  const moderateIssues =
    peaks.filter(p => p.severity === 'moderate').length +
    nulls.filter(n => n.severity === 'moderate').length;

  let severity: Severity = 'negligible';
  if (significantIssues >= 3) severity = 'significant';
  else if (significantIssues >= 1 || moderateIssues >= 4) severity = 'moderate';
  else if (moderateIssues >= 1) severity = 'minor';

  return {
    data: {
      peaks,
      nulls,
      sbir_nulls: sbirNulls,
      glm_addressable_peaks: glmAddressablePeaks,
      unaddressable_nulls: unaddressableNulls
    },
    summary,
    recommendations,
    severity,
    confidence: 'high'
  };
}
```
