```typescript
import type { ConfidenceLevel, Severity } from '../types/index.js';
import type {
  SubIntegrationAnalysis,
  CrossoverAnalysis,
  TimingRecommendation,
  PolarityRecommendation
} from '../analysis/sub-integration.js';
import type { InterpretedResult, Recommendation } from './types.js';

export interface PhaseInversionDetection {
  is_phase_inverted: boolean;
  phase_difference_deg: number;
  inversion_recommended: boolean;
  expected_improvement_db: number;
  confidence: ConfidenceLevel;
  explanation: string;
}

export interface SubIntegrationData {
  crossover: CrossoverAnalysis;
  timing: TimingRecommendation;
  polarity: PolarityRecommendation;
  phase_inversion: PhaseInversionDetection;
  summation_dip_db: number;
  predicted_improvement_db: number;
}

/**
 * Detect if sub is phase inverted relative to mains
 * Phase inversion occurs when phase difference is near 180 degrees (150-210 range)
 */
export function detectPhaseInversion(
  crossover: CrossoverAnalysis,
  polarity: PolarityRecommendation
): PhaseInversionDetection {
  const phaseDiff = Math.abs(crossover.phase_difference_deg);

  // Phase inversion range: 150-210 degrees (near 180)
  const isNear180 = phaseDiff >= 150 && phaseDiff <= 210;

  // Also consider if near 180 after accounting for full rotation
  const adjustedDiff = phaseDiff > 180 ? 360 - phaseDiff : phaseDiff;
  const isInverted = adjustedDiff >= 150 && adjustedDiff <= 210;

  if (!isNear180 && !isInverted) {
    return {
      is_phase_inverted: false,
      phase_difference_deg: crossover.phase_difference_deg,
      inversion_recommended: false,
      expected_improvement_db: 0,
      confidence: 'high',
      explanation: `Phase difference is ${crossover.phase_difference_deg.toFixed(0)} degrees - not near 180 degrees, no inversion needed`
    };
  }

  // Use polarity analysis results for improvement estimate
  const shouldInvert = polarity.invert_recommended;
  const improvement = polarity.expected_improvement_db;

  return {
    is_phase_inverted: true,
    phase_difference_deg: crossover.phase_difference_deg,
    inversion_recommended: shouldInvert,
    expected_improvement_db: improvement,
    confidence: improvement > 3 ? 'high' : 'medium',
    explanation: shouldInvert
      ? `Phase difference is ${crossover.phase_difference_deg.toFixed(0)} degrees (near 180) - inverting sub polarity should improve summing by ${improvement.toFixed(1)} dB`
      : `Phase difference is ${crossover.phase_difference_deg.toFixed(0)} degrees but inversion would not significantly improve response`
  };
}

export function interpretSubIntegration(
  analysis: SubIntegrationAnalysis
): InterpretedResult<SubIntegrationData> {
  const { crossover_analysis, timing_recommendations, polarity_recommendation, summation_prediction } = analysis;

  // Detect phase inversion
  const phaseInversion = detectPhaseInversion(crossover_analysis, polarity_recommendation);

  // Build summary
  const summaryParts: string[] = [];

  // Crossover summary
  summaryParts.push(
    `Crossover detected at ${crossover_analysis.detected_crossover_hz.toFixed(0)} Hz with ${crossover_analysis.phase_alignment_quality} phase alignment`
  );

  // Phase difference summary
  if (crossover_analysis.phase_alignment_quality === 'poor') {
    summaryParts.push(
      `Phase difference of ${crossover_analysis.phase_difference_deg.toFixed(0)} degrees is causing summing issues`
    );
  }

  // Timing summary
  if (Math.abs(timing_recommendations.adjustment_needed_ms) > 1) {
    summaryParts.push(
      `Subwoofer timing needs ${timing_recommendations.adjustment_needed_ms.toFixed(1)} ms adjustment`
    );
  } else {
    summaryParts.push('Subwoofer timing is well aligned');
  }

  // Phase inversion summary
  if (phaseInversion.is_phase_inverted && phaseInversion.inversion_recommended) {
    summaryParts.push(
      `Phase inversion detected - flipping sub polarity should improve by ${phaseInversion.expected_improvement_db.toFixed(1)} dB`
    );
  }

  // Summation summary
  if (summation_prediction.current_dip_at_crossover_db > 6) {
    summaryParts.push(
      `Significant dip of ${summation_prediction.current_dip_at_crossover_db.toFixed(1)} dB at crossover`
    );
  }

  const summary = summaryParts.join('. ') + '.';

  // Generate recommendations
  const recommendations: Recommendation[] = [];

  // Phase inversion is highest priority (settings fix)
  if (phaseInversion.inversion_recommended) {
    recommendations.push({
      action: `Invert subwoofer polarity (flip phase 180 degrees)`,
      expected_impact: `Should improve crossover summing by ${phaseInversion.expected_improvement_db.toFixed(1)} dB`,
      priority: 1,
      fixability: 'settings',
      category: 'sub_integration'
    });
  }

  // Timing adjustment
  if (Math.abs(timing_recommendations.adjustment_needed_ms) > 1) {
    const direction = timing_recommendations.adjustment_needed_ms > 0 ? 'Increase' : 'Decrease';
    recommendations.push({
      action: `${direction} sub delay by ${Math.abs(timing_recommendations.adjustment_needed_ms).toFixed(1)} ms`,
      expected_impact: 'Should improve phase alignment at crossover',
      priority: phaseInversion.inversion_recommended ? 2 : 1,
      fixability: 'settings',
      category: 'sub_integration'
    });
  }

  // Position recommendation if phase alignment is poor and settings won't fix it
  if (crossover_analysis.phase_alignment_quality === 'poor' &&
      !phaseInversion.inversion_recommended &&
      Math.abs(timing_recommendations.adjustment_needed_ms) <= 5) {
    recommendations.push({
      action: 'Reposition subwoofer to improve phase alignment with mains',
      expected_impact: 'May significantly improve crossover integration',
      priority: 1,
      fixability: 'placement',
      category: 'sub_integration'
    });
  }

  // Determine severity
  let severity: Severity = 'negligible';
  if (summation_prediction.current_dip_at_crossover_db > 10 ||
      crossover_analysis.phase_alignment_quality === 'poor') {
    severity = 'significant';
  } else if (summation_prediction.current_dip_at_crossover_db > 6 ||
             crossover_analysis.phase_alignment_quality === 'fair') {
    severity = 'moderate';
  } else if (summation_prediction.current_dip_at_crossover_db > 3) {
    severity = 'minor';
  }

  return {
    data: {
      crossover: crossover_analysis,
      timing: timing_recommendations,
      polarity: polarity_recommendation,
      phase_inversion: phaseInversion,
      summation_dip_db: summation_prediction.current_dip_at_crossover_db,
      predicted_improvement_db: summation_prediction.improvement_db
    },
    summary,
    recommendations,
    severity,
    confidence: analysis.confidence
  };
}
```

```typescript
import type { FrequencyResponseData, ConfidenceLevel, Severity } from '../types/index.js';
import type { InterpretedResult, Recommendation } from './types.js';

export type SymmetryRating = 'excellent' | 'good' | 'fair' | 'poor';

export interface BandSymmetry {
  band_name: string;
  frequency_range_hz: [number, number];
  left_avg_db: number;
  right_avg_db: number;
  level_deviation_db: number;
  left_variance_db: number;
  right_variance_db: number;
  variance_deviation_db: number;
  rating: SymmetryRating;
  impact_on_imaging: 'none' | 'minor' | 'moderate' | 'significant';
}

export interface LRSymmetryData {
  overall_rating: SymmetryRating;
  overall_deviation_db: number;
  band_analysis: BandSymmetry[];
  asymmetric_bands: string[];
}

// Thresholds from RESEARCH.md: <1 dB = excellent, 1-2 dB = good, 2-3 dB = fair, >3 dB = poor
function rateDeviation(levelDev: number, _varianceDev: number): SymmetryRating {
  const absLevel = Math.abs(levelDev);
  if (absLevel < 1) return 'excellent';
  if (absLevel < 2) return 'good';
  if (absLevel < 3) return 'fair';
  return 'poor';
}

function assessImagingImpact(levelDev: number, varianceDev: number): 'none' | 'minor' | 'moderate' | 'significant' {
  const absLevel = Math.abs(levelDev);
  const absVariance = Math.abs(varianceDev);

  // Thresholds from RESEARCH.md
  if (absLevel < 1 && absVariance < 2) return 'none';
  if (absLevel < 2 && absVariance < 4) return 'minor';
  if (absLevel < 3 && absVariance < 6) return 'moderate';
  return 'significant';
}

function calculateBandStats(
  frequencies: number[],
  spl: number[],
  minFreq: number,
  maxFreq: number
): { avg: number; variance: number } {
  const values: number[] = [];
  for (let i = 0; i < frequencies.length; i++) {
    if (frequencies[i] >= minFreq && frequencies[i] <= maxFreq) {
      values.push(spl[i]);
    }
  }

  if (values.length === 0) return { avg: 0, variance: 0 };

  const avg = values.reduce((a, b) => a + b, 0) / values.length;
  const max = Math.max(...values);
  const min = Math.min(...values);
  const variance = max - min;

  return { avg, variance };
}

export function interpretLRSymmetry(
  left: FrequencyResponseData,
  right: FrequencyResponseData
): InterpretedResult<LRSymmetryData> {
  // Frequency bands to analyze
  const bands: Array<{ name: string; range: [number, number] }> = [
    { name: 'Bass', range: [60, 200] },
    { name: 'Midrange', range: [200, 2000] },
    { name: 'Upper Midrange', range: [2000, 6000] },
    { name: 'Treble', range: [6000, 20000] }
  ];

  const bandAnalysis: BandSymmetry[] = [];
  const asymmetricBands: string[] = [];
  let totalDeviation = 0;
  let bandCount = 0;

  for (const band of bands) {
    const leftStats = calculateBandStats(left.frequencies_hz, left.spl_db, band.range[0], band.range[1]);
    const rightStats = calculateBandStats(right.frequencies_hz, right.spl_db, band.range[0], band.range[1]);

    const levelDev = leftStats.avg - rightStats.avg;
    const varianceDev = leftStats.variance - rightStats.variance;
    const rating = rateDeviation(levelDev, varianceDev);
    const impact = assessImagingImpact(levelDev, varianceDev);

    bandAnalysis.push({
      band_name: band.name,
      frequency_range_hz: band.range,
      left_avg_db: Math.round(leftStats.avg * 10) / 10,
      right_avg_db: Math.round(rightStats.avg * 10) / 10,
      level_deviation_db: Math.round(levelDev * 10) / 10,
      left_variance_db: Math.round(leftStats.variance * 10) / 10,
      right_variance_db: Math.round(rightStats.variance * 10) / 10,
      variance_deviation_db: Math.round(varianceDev * 10) / 10,
      rating,
      impact_on_imaging: impact
    });

    if (rating === 'fair' || rating === 'poor') {
      asymmetricBands.push(band.name);
    }

    totalDeviation += Math.abs(levelDev);
    bandCount++;
  }

  // Overall rating based on average deviation and worst band
  const avgDeviation = bandCount > 0 ? totalDeviation / bandCount : 0;
  const worstRating = bandAnalysis.reduce((worst, band) => {
    const ratings: SymmetryRating[] = ['excellent', 'good', 'fair', 'poor'];
    return ratings.indexOf(band.rating) > ratings.indexOf(worst) ? band.rating : worst;
  }, 'excellent' as SymmetryRating);

  // Overall rating is blend of average and worst
  let overallRating: SymmetryRating;
  if (avgDeviation < 1 && worstRating !== 'poor') {
    overallRating = 'excellent';
  } else if (avgDeviation < 2 && worstRating !== 'poor') {
    overallRating = 'good';
  } else if (avgDeviation < 3 || worstRating === 'fair') {
    overallRating = 'fair';
  } else {
    overallRating = 'poor';
  }

  // Generate summary
  let summary = `L/R symmetry is ${overallRating} with average deviation of ${avgDeviation.toFixed(1)} dB. `;

  if (asymmetricBands.length === 0) {
    summary += 'All frequency bands are well matched.';
  } else if (asymmetricBands.length === 1) {
    summary += `${asymmetricBands[0]} band shows asymmetry that may affect imaging.`;
  } else {
    summary += `${asymmetricBands.join(' and ')} bands show asymmetry that may affect imaging.`;
  }

  // Generate recommendations
  const recommendations: Recommendation[] = [];

  for (const band of bandAnalysis) {
    if (band.rating === 'poor') {
      const direction = band.level_deviation_db > 0 ? 'Left is louder' : 'Right is louder';
      recommendations.push({
        action: `Check ${band.band_name.toLowerCase()} response asymmetry (${direction} by ${Math.abs(band.level_deviation_db).toFixed(1)} dB)`,
        expected_impact: 'Improving symmetry will tighten stereo imaging',
        priority: 1,
        fixability: 'placement',
        category: 'lr_symmetry'
      });
    } else if (band.rating === 'fair') {
      recommendations.push({
        action: `Consider addressing ${band.band_name.toLowerCase()} asymmetry (${Math.abs(band.level_deviation_db).toFixed(1)} dB difference)`,
        expected_impact: 'May improve stereo imaging and phantom center',
        priority: 2,
        fixability: 'placement',
        category: 'lr_symmetry'
      });
    }
  }

  // Add general recommendation if significant asymmetry
  if (overallRating === 'poor' || overallRating === 'fair') {
    recommendations.push({
      action: 'Verify speaker placement symmetry and check for room asymmetries (furniture, openings)',
      expected_impact: 'Addressing root cause of asymmetry',
      priority: recommendations.length > 0 ? 3 : 1,
      fixability: 'placement',
      category: 'lr_symmetry'
    });
  }

  // Determine severity
  let severity: Severity = 'negligible';
  if (overallRating === 'poor') severity = 'significant';
  else if (overallRating === 'fair') severity = 'moderate';
  else if (overallRating === 'good' && asymmetricBands.length > 0) severity = 'minor';

  return {
    data: {
      overall_rating: overallRating,
      overall_deviation_db: Math.round(avgDeviation * 10) / 10,
      band_analysis: bandAnalysis,
      asymmetric_bands: asymmetricBands
    },
    summary,
    recommendations,
    severity,
    confidence: 'high'
  };
}
```
