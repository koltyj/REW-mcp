import { describe, it, expect, beforeEach, vi } from 'vitest';
import { readSessionResource, listSessionResources } from './session-resource.js';
import { createSession, clearAllSessions } from '../session/index.js';

describe('session-resource', () => {
  beforeEach(() => {
    clearAllSessions();
  });

  describe('readSessionResource', () => {
    it('returns session state for valid session ID', () => {
      const session = createSession('Test session');
      const result = readSessionResource(session.session_id);
      expect(result.session_id).toBe(session.session_id);
      expect(result.sequence_step).toBe('idle');
    });

    it('throws -32002 error for invalid session ID', () => {
      const invalidId = '00000000-0000-0000-0000-000000000000';
      try {
        readSessionResource(invalidId);
        expect.fail('Should have thrown');
      } catch (error: any) {
        expect(error.code).toBe(-32002);
        expect(error.message).toContain('not found');
      }
    });
  });

  describe('listSessionResources', () => {
    it('returns empty array when no sessions', () => {
      expect(listSessionResources()).toEqual([]);
    });

    it('returns resource objects for active sessions', () => {
      createSession('Session 1');
      createSession('Session 2');
      const resources = listSessionResources();
      expect(resources).toHaveLength(2);
      expect(resources[0].uri).toMatch(/^session:\/\//);
      expect(resources[0].mimeType).toBe('application/json');
    });
  });
});

import { describe, it, expect, beforeEach } from 'vitest';
import { MEASUREMENT_WORKFLOW_PROMPT, getMeasurementWorkflowMessages } from './measurement-workflow.js';
import { createSession, clearAllSessions } from '../session/index.js';

describe('measurement-workflow prompt', () => {
  beforeEach(() => {
    clearAllSessions();
  });

  it('requires session_id argument', () => {
    const sessionArg = MEASUREMENT_WORKFLOW_PROMPT.arguments?.find(a => a.name === 'session_id');
    expect(sessionArg?.required).toBe(true);
  });

  it('includes embedded session resource for valid session', () => {
    const session = createSession();
    const messages = getMeasurementWorkflowMessages({ session_id: session.session_id });

    // Should have user message + assistant resource message
    expect(messages.length).toBeGreaterThanOrEqual(1);

    // Find the resource message
    const resourceMsg = messages.find(m => m.content?.type === 'resource');
    if (resourceMsg) {
      expect(resourceMsg.content.resource.uri).toMatch(/^session:\/\//);
    }
  });

  it('throws for invalid session_id', () => {
    expect(() => getMeasurementWorkflowMessages({ session_id: 'invalid' })).toThrow();
  });
});
