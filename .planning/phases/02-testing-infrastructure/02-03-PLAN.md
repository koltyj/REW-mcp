/**
 * Tests for rew.api_audio tool
 */

import { describe, it, expect, beforeEach, afterEach } from 'vitest';
import { http, HttpResponse } from 'msw';
import { setupServer } from 'msw/node';
import { executeApiAudio, ApiAudioInputSchema } from './api-audio.js';
import { setActiveApiClient, clearActiveApiClient } from './api-connect.js';
import { REWApiClient } from '../api/rew-client.js';

const server = setupServer();

describe('rew.api_audio tool', () => {
  let client: REWApiClient;

  beforeEach(async () => {
    server.listen({ onUnhandledRequest: 'warn' });

    // Mock connect endpoint
    server.use(
      http.get('http://127.0.0.1:4735/doc.json', () => {
        return HttpResponse.json({ info: { version: '5.30.9' } });
      })
    );

    client = new REWApiClient();
    await client.connect();
    setActiveApiClient(client);
  });

  afterEach(() => {
    clearActiveApiClient();
    server.resetHandlers();
    server.close();
  });

  describe('input validation', () => {
    it('should validate action enum', () => {
      expect(() => ApiAudioInputSchema.parse({ action: 'invalid' })).toThrow();
      expect(() => ApiAudioInputSchema.parse({ action: 'status' })).not.toThrow();
    });
  });

  describe('status action', () => {
    it('should return audio status', async () => {
      server.use(
        http.get('http://127.0.0.1:4735/application/audio', () => {
          return HttpResponse.json({ data: { enabled: true, ready: true } });
        }),
        http.get('http://127.0.0.1:4735/application/audio-driver', () => {
          return HttpResponse.json({ data: 'CoreAudio' });
        }),
        http.get('http://127.0.0.1:4735/application/sample-rate', () => {
          return HttpResponse.json({ data: 48000 });
        }),
        http.get('http://127.0.0.1:4735/application/java-input-device', () => {
          return HttpResponse.json({ data: 'Built-in Microphone' });
        }),
        http.get('http://127.0.0.1:4735/application/java-output-device', () => {
          return HttpResponse.json({ data: 'Built-in Output' });
        }),
        http.get('http://127.0.0.1:4735/application/input-calibration', () => {
          return HttpResponse.json({ data: { enabled: false } });
        })
      );

      const result = await executeApiAudio({ action: 'status' });

      expect(result.status).toBe('success');
      expect(result.data?.audio_status?.enabled).toBe(true);
      expect(result.data?.audio_status?.ready).toBe(true);
    });
  });

  describe('list_devices action', () => {
    it('should list available devices', async () => {
      server.use(
        http.get('http://127.0.0.1:4735/application/java-input-devices', () => {
          return HttpResponse.json({ data: ['Mic 1', 'Mic 2'] });
        }),
        http.get('http://127.0.0.1:4735/application/java-output-devices', () => {
          return HttpResponse.json({ data: ['Speaker 1', 'Speaker 2'] });
        }),
        http.get('http://127.0.0.1:4735/application/sample-rates', () => {
          return HttpResponse.json({ data: [44100, 48000, 96000] });
        })
      );

      const result = await executeApiAudio({ action: 'list_devices' });

      expect(result.status).toBe('success');
      expect(result.data?.available_devices?.input_devices).toContain('Mic 1');
      expect(result.data?.available_devices?.output_devices).toContain('Speaker 1');
    });
  });

  describe('set_input action', () => {
    it('should set input device', async () => {
      server.use(
        http.post('http://127.0.0.1:4735/application/java-input-device', () => {
          return HttpResponse.json({ status: 200 });
        })
      );

      const result = await executeApiAudio({ action: 'set_input', device: 'Mic 1' });

      expect(result.status).toBe('success');
    });

    it('should require device parameter', async () => {
      const result = await executeApiAudio({ action: 'set_input' });

      expect(result.status).toBe('error');
      expect(result.message).toContain('device');
    });
  });

  describe('error handling', () => {
    it('should handle not connected', async () => {
      clearActiveApiClient();

      const result = await executeApiAudio({ action: 'status' });

      expect(result.status).toBe('error');
      expect(result.error_type).toBe('connection_error');
    });
  });
});

/**
 * Tests for rew.api_measure tool
 */

import { describe, it, expect, beforeEach, afterEach } from 'vitest';
import { http, HttpResponse } from 'msw';
import { setupServer } from 'msw/node';
import { executeApiMeasure, ApiMeasureInputSchema } from './api-measure.js';
import { setActiveApiClient, clearActiveApiClient } from './api-connect.js';
import { REWApiClient } from '../api/rew-client.js';

const server = setupServer();

describe('rew.api_measure tool', () => {
  let client: REWApiClient;

  beforeEach(async () => {
    server.listen({ onUnhandledRequest: 'warn' });

    server.use(
      http.get('http://127.0.0.1:4735/doc.json', () => {
        return HttpResponse.json({ info: { version: '5.30.9' } });
      })
    );

    client = new REWApiClient();
    await client.connect();
    setActiveApiClient(client);
  });

  afterEach(() => {
    clearActiveApiClient();
    server.resetHandlers();
    server.close();
  });

  describe('input validation', () => {
    it('should validate action enum', () => {
      expect(() => ApiMeasureInputSchema.parse({ action: 'invalid' })).toThrow();
      expect(() => ApiMeasureInputSchema.parse({ action: 'start' })).not.toThrow();
    });
  });

  describe('start action', () => {
    it('should start measurement with defaults', async () => {
      server.use(
        http.post('http://127.0.0.1:4735/measurements/sweep', () => {
          return HttpResponse.json({
            data: { uuid: 'new-uuid', name: 'New Measurement' }
          });
        })
      );

      const result = await executeApiMeasure({ action: 'start' });

      expect(result.status).toBe('success');
      expect(result.data?.measurement_uuid).toBe('new-uuid');
    });

    it('should configure sweep parameters', async () => {
      let capturedBody: unknown;
      server.use(
        http.post('http://127.0.0.1:4735/measurements/sweep', async ({ request }) => {
          capturedBody = await request.json();
          return HttpResponse.json({
            data: { uuid: 'new-uuid', name: 'Custom Sweep' }
          });
        })
      );

      const result = await executeApiMeasure({
        action: 'start',
        start_freq_hz: 20,
        end_freq_hz: 20000,
        level_dbfs: -18
      });

      expect(result.status).toBe('success');
      expect(capturedBody).toMatchObject({
        startFrequency: 20,
        endFrequency: 20000
      });
    });
  });

  describe('error handling', () => {
    it('should handle not connected', async () => {
      clearActiveApiClient();

      const result = await executeApiMeasure({ action: 'start' });

      expect(result.status).toBe('error');
      expect(result.error_type).toBe('connection_error');
    });

    it('should handle measurement failure', async () => {
      server.use(
        http.post('http://127.0.0.1:4735/measurements/sweep', () => {
          return HttpResponse.json({ error: 'Audio not ready' }, { status: 500 });
        })
      );

      const result = await executeApiMeasure({ action: 'start' });

      expect(result.status).toBe('error');
    });
  });
});
